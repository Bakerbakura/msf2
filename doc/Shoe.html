def Shoe.sizeToPreSize(_size, _sizeType, _lengthFit)
      interval = Sizetypes.find_by_SizeType!(_sizeType).SizeTypeInterval

      return case _lengthFit
              when "Too Short"                    then ToMondo(_size + 0.4*interval,       _sizeType)
              when "Slightly Short"       then ToMondo(_size + 0.2*interval,    _sizeType)
              when "Perfect"                              then ToMondo(_size,                                                                       _sizeType)
              when "Slightly Long"        then ToMondo(_size - 0.2*interval,     _sizeType)
              when "Too Long"                             then ToMondo(_size - 0.4*interval,       _sizeType)
              else "INVALID _LENGTHFIT!"
      end
end

def FromMondo(_mondoSize, _sizeType)
      st = Sizetypes.find_by_Sizetype!(_sizeType)
      return (_mondoSize - st.ToMondo0)/st.ToMondo1
end

def ToMondo(_size, _sizeType)
      st = Sizetypes.find_by_SizeType!(_sizeType)
      return _size*st.ToMondo1 + st.ToMondo0
end

def preToRealSize(_t2rsid, _preSize)
      t2rs = Typetorealsize.find_by!(T2RS_ID: _t2rsid)
      return _preSize*t2rs.ToMondo1 + t2rs.ToMondo0
end
